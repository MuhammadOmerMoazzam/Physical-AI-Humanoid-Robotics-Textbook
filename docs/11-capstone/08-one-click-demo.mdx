# THREE TERMINALS. THAT'S IT.

```bash
# Terminal 1 – Isaac Sim (or real robot)
./isaac-sim.sh capstone/isaac_sim_scenes/kitchen_conversational_humanoid.usd

# Terminal 2 – Full ROS 2 stack
ros2 launch capstone_bringup capstone_full.launch.py use_sim:=true

# Terminal 3 – Speak naturally
python speak_to_robot.py
# → "Good morning. Please clean the table and bring me a cold water."
```
